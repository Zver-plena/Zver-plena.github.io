懒汉式创建方法

普通懒汉式代码实现
class Person{
    private static Person instance=null;
    public static Person getInstance(){
        if(instance==null){
            instance=new person();
        }
        return instance;
    }
}
多线程安全懒汉式代码实现
class Person {
    private static Person instance=null;
    public static Person getInstance(){
        if(instance==null){
            synchronized(Person.class){
                instance=new Person();
            }
        }
        return instance;
    }
}
