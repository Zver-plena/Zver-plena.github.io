json

package main

import (
	"encoding/json"
	"fmt"
)

type userInfo struct {
	Name  string
	Age   int `json:"age"`
	Hobby []string
}

func main() {
	a := userInfo{Name: "wang", Age: 18, Hobby: []string{"Golang", "TypeScript"}}
	//转换成json格式，序列化
	buf, err := json.Marshal(a)
	if err != nil {
		panic(err)
	}
	//直接打印：16进制
	fmt.Println(buf) // [123 34 78 97...]
	//string强转换
	fmt.Println(string(buf)) // {"Name":"wang","age":18,"Hobby":["Golang","TypeScript"]}

	//json格式化
	buf, err = json.MarshalIndent(a, "", "\t")
	if err != nil {
		panic(err)
	}
	fmt.Println(string(buf))

	var b userInfo
	//反序列化
	err = json.Unmarshal(buf, &b)
	if err != nil {
		panic(err)
	}
	//# 打印具体的类型
	fmt.Printf("%#v\n", b) // main.userInfo{Name:"wang", Age:18, Hobby:[]string{"Golang", "TypeScript"}}
}

package main

import (
	"errors"
	"fmt"
)

type user struct {
	name     string
	password string
}

//函数可以向外抛出error
func findUser(users []user, name string) (v *user, err error) {
	for _, u := range users {
		if u.name == name {
			return &u, nil
		}
	}
	//抛出错误
	return nil, errors.New("not found")
}

func main() {
	u, err := findUser([]user{{"wang", "1024"}}, "wang")
	if err != nil {
		//直接打印错误
		fmt.Println(err)
		return
	}
	fmt.Println(u.name) // wang

	if u, err := findUser([]user{{"wang", "1024"}}, "li"); err != nil {
		fmt.Println(err) // not found
		return
	} else {
		fmt.Println(u.name)
	}
}

strconv

package main

import (
	"fmt"
	"strconv"
)

func main() {

	//转换成浮点数
	f, _ := strconv.ParseFloat("1.234", 64)
	fmt.Println(f) // 1.234
	//字符串转换成整型，base：转换的进制 ： 如果不符合要转换的进制的要求为 0
	//例如 如果要转换成2进制，字符串中却出现了3，4。。。
	n, _ := strconv.ParseInt("1111", 2, 64)
	fmt.Println(n) // 111
	//进制转换, 0：自动识别字符串内的进制，如果没有说明按照10进制处理，最后转换成10进制
	n, _ = strconv.ParseInt("0x0010", 0, 64)
	fmt.Println(n) // 4096
	//字符串转换成整型
	n2, _ := strconv.Atoi("123")
	fmt.Println(n2) // 123

	n2, err := strconv.Atoi("AAA")
	fmt.Println(n2, err) // 0 strconv.Atoi: parsing "AAA": invalid syntax
}

package main

import (
   "fmt"
   "time"
)

func main() {
   now := time.Now()
   //获取时间
   fmt.Println(now) // 2022-03-27 18:04:59.433297 +0800 CST m=+0.000087933
   //可以自己显示设置时间和时区
   t := time.Date(2022, 3, 27, 1, 25, 36, 0, time.UTC)
   t2 := time.Date(2022, 3, 27, 2, 30, 36, 0, time.UTC)
   fmt.Println(t) // 2022-03-27 01:25:36 +0000 UTC
   //也可以通过.选择性输出
   fmt.Println(t.Year(), t.Month(), t.Day(), t.Hour(), t.Minute()) // 2022 March 27 1 25
   //时间格式化
   fmt.Println(t.Format("2006-01-02 15:04:05")) // 2022-03-27 01:25:36
   //通过.Sub函数可以得出两个时间之间的时间差
   diff := t2.Sub(t)
   fmt.Println(diff) // 1h5m0s
   //时间差可以转换成分，秒
   fmt.Println(diff.Minutes(), diff.Seconds()) // 65 3900
   //时间格式判断
   t3, err := time.Parse("2006-01-02 15:04:05", "2022-03-27 01:25:36")
   if err != nil {
      panic(err)
   }
   fmt.Println(t3 == t) // true
   //获取时间戳
   fmt.Println(now.Unix()) // 1648738080
}

工作流程

具体例子代码

package main

import (
	"bufio"
	"context"
	"encoding/binary"
	"errors"
	"fmt"
	"io"
	"log"
	"net"
)

const socks5Ver = 0x05
const cmdBind = 0x01
const atypIPV4 = 0x01
const atypeHOST = 0x03
const atypeIPV6 = 0x04

func main() {
	//监听端口
	server, err := net.Listen("tcp", "127.0.0.1:1080")
	if err != nil {
		panic(err)
	}
	for {
		//开启接受连接的状态
		client, err := server.Accept()
		if err != nil {
			log.Printf("Accept failed %v", err)
			continue
		}
		//处理连接
		go process(client)
	}
}

func process(conn net.Conn) {
	defer conn.Close()
	//浏览器向代理服务器的发送请求
	reader := bufio.NewReader(conn) //从连接中创建缓冲的只读流
	//1.协商阶段，验证浏览器的请求
	err := auth(reader, conn)
	if err != nil {
		log.Printf("client %v auth failed:%v", conn.RemoteAddr(), err)
		return
	}

	//2.对浏览器发送的请求进行处理 + 3.发送数据readder
	err = connect(reader, conn)
	if err != nil {
		log.Printf("client %v auth failed:%v", conn.RemoteAddr(), err)
		return
	}
}

//验证浏览器请求模块(验证报文)
func auth(reader *bufio.Reader, conn net.Conn) (err error) {
	// +----+----------+----------+
	// |VER | NMETHODS | METHODS  |
	// +----+----------+----------+
	// | 1  |    1     | 1 to 255 |
	// +----+----------+----------+
	// VER: 协议版本，socks5为0x05
	// NMETHODS: 支持认证的方法数量
	// METHODS: 对应NMETHODS，NMETHODS的值为多少，METHODS就有多少个字节。RFC预定义了一些值的含义，内容如下:
	// X’00’ NO AUTHENTICATION REQUIRED
	// X’02’ USERNAME/PASSWORD

	////读取协议版本
	ver, err := reader.ReadByte()
	if err != nil {
		return fmt.Errorf("read ver failed:%w", err)
	}
	if ver != socks5Ver {
		return fmt.Errorf("not supported ver:%v", ver)
	}
	//读取方法数量
	methodSize, err := reader.ReadByte()
	if err != nil {
		return fmt.Errorf("read methodSize failed:%w", err)
	}
	method := make([]byte, methodSize)
	//填充方法
	_, err = io.ReadFull(reader, method)
	if err != nil {
		return fmt.Errorf("read method failed:%w", err)
	}

	// +----+--------+
	// |VER | METHOD |
	// +----+--------+
	// | 1  |   1    |
	// +----+--------+

	//1.1 通过验证，代理服务器的认证响应
	_, err = conn.Write([]byte{socks5Ver, 0x00})
	if err != nil {
		return fmt.Errorf("write failed:%w", err)
	}
	return nil
}

//验证成功后，就建立了连接
//浏览器把需要执行的操作指令发给代理服务器，表明自己要执行代理的请求。请求帧格式
func connect(reader *bufio.Reader, conn net.Conn) (err error) {
	// +----+-----+-------+------+----------+----------+
	// |VER | CMD |  RSV  | ATYP | DST.ADDR | DST.PORT |
	// +----+-----+-------+------+----------+----------+
	// | 1  |  1  | X'00' |  1   | Variable |    2     |
	// +----+-----+-------+------+----------+----------+
	// VER 版本号，socks5的值为0x05
	// CMD 0x01表示CONNECT请求
	// RSV 保留字段，值为0x00
	// ATYP 目标地址类型，DST.ADDR的数据对应这个字段的类型。
	//   0x01表示IPv4地址，DST.ADDR为4个字节
	//   0x03表示域名，DST.ADDR是一个可变长度的域名
	// DST.ADDR 一个可变长度的值
	// DST.PORT 目标端口，固定2个字节

	// 读取请求帧
	buf := make([]byte, 4)
	_, err = io.ReadFull(reader, buf)
	if err != nil {
		return fmt.Errorf("read header failed:%w", err)
	}
	ver, cmd, atyp := buf[0], buf[1], buf[3]
	//判断socks版本
	if ver != socks5Ver {
		return fmt.Errorf("not supported ver:%v", ver)
	}
	//判断请求
	if cmd != cmdBind {
		return fmt.Errorf("not supported cmd:%v", ver)
	}
	//获取目标服务器地址 addr
	addr := ""
	switch atyp {
	case atypIPV4:
		//如果是IPV4，容量为4，正好可以用之前的buf，填充到buf里面，这里其实就是覆盖了之前的数据
		//因为之前的数据已经取出来到变量了。重复利用
		_, err = io.ReadFull(reader, buf)
		if err != nil {
			return fmt.Errorf("read atyp failed:%w", err)
		}
		addr = fmt.Sprintf("%d.%d.%d.%d", buf[0], buf[1], buf[2], buf[3])
	case atypeHOST:
		//如果是HOST，需要先读取他的长度
		hostSize, err := reader.ReadByte()
		if err != nil {
			return fmt.Errorf("read hostSize failed:%w", err)
		}
		//创建对应长度的map
		host := make([]byte, hostSize)
		//填充map
		_, err = io.ReadFull(reader, host)
		if err != nil {
			return fmt.Errorf("read host failed:%w", err)
		}
		addr = string(host)
	case atypeIPV6:
		return errors.New("IPv6: no supported yet")
	default:
		return errors.New("invalid atyp")
	}
	_, err = io.ReadFull(reader, buf[:2])
	if err != nil {
		return fmt.Errorf("read port failed:%w", err)
	}

	//获得目标端口 port
	port := binary.BigEndian.Uint16(buf[:2]) //填充两个字节到buf，覆盖之前的，重复利用

	//2.1 代理服务器 和 目标服务器 建立tcp连接

	dest, err := net.Dial("tcp", fmt.Sprintf("%v:%v", addr, port)) //用addr和port 和目标服务器建立tcp连接 -> dest

	//2.2 目标服务器 响应给 代理服务器 成功建立tcp连接
	if err != nil {
		return fmt.Errorf("dial dst failed:%w", err)
	}
	defer dest.Close()

	log.Println("dial", addr, port) // 打印目标服务器的addr和port

	// +----+-----+-------+------+----------+----------+
	// |VER | REP |  RSV  | ATYP | BND.ADDR | BND.PORT |
	// +----+-----+-------+------+----------+----------+
	// | 1  |  1  | X'00' |  1   | Variable |    2     |
	// +----+-----+-------+------+----------+----------+
	// VER socks版本，这里为0x05
	// REP Relay field,内容取值如下 X’00’ succeeded
	// RSV 保留字段
	// ATYPE 地址类型
	// BND.ADDR 服务绑定的地址
	// BND.PORT 服务绑定的端口DST.PORT

	// 2.3 代理服务器 响应给 浏览器 成功和目标服务器建立连接
	_, err = conn.Write([]byte{0x05, 0x00, 0x00, 0x01, 0, 0, 0, 0, 0, 0}) //响应报文给浏览器，连接的情况
	if err != nil {
		return fmt.Errorf("write failed: %w", err)
	}

	ctx, cancel := context.WithCancel(context.Background())
	
	//直到传输出现问题,执行了cancel，否则ctx.Done()会一直堵塞
	defer cancel()

	//copy(to , from )
	//	      <-

	//实现信息转发

	//3.1 relay数据

	go func() {
		_, _ = io.Copy(dest, reader) //浏览器发送数据给服务器，死循环
		cancel()
	}()
	//3.2 响应结果 + 3.3. 响应结果
	go func() {
		_, _ = io.Copy(conn, dest) //服务器发送数据给浏览器
		cancel()
	}()

	<-ctx.Done()
	return nil
}

Golang工程基础

并发

Goroutine

go从底层原生支持了goroutine来创建协程来实现并发操作

CSP（commuication Sequential Processes）

go在多个运行的goroutine中提倡通过channel来实现共享内存

而不是通过共享内存来实现通信

channel

make (chan 元素类型,[缓冲大小])

  • make(char int)：不带缓冲
  • make(char int ,2)：有缓冲区

通过channel实现gorountine之间的通信

package concurrence

func CalSquare() {
	src := make(chan int)
	dest := make(chan int, 3)
	go func() {
		defer close(src)
		for i := 0; i < 10; i++ {
			src <- i
		}
	}()
	go func() {
		defer close(dest)
		for i := range src {
			dest <- i * i
		}
	}()
	for i := range dest {
		println(i)
	}
}

WaitGroup

因为goroutine启动没有什么负担，但是主线程结束的时候，goroutine会被强制结束

所以go提供了sync.WaitGroup来进行主线程和goroutine之间的通信，确保所有的goroutine执行完成之后才结束主线程

func ManyGoWait() {
	var wg sync.WaitGroup
	// 计数 ： 表示有5个并发任务
	wg.Add(5)
	for i := 0; i < 5; i++ {
		go func(j int) {
			// 每一次 wg.Done() 计数--
			defer wg.Done()
			hello(j)
		}(i)
	}
	//如果计数不为0，会阻塞，直到计数为0
	wg.Wait()
}

并发安全问题Lock

当操作共享的数据的时候，并发执行会造成死锁，脏数据

go这里提供了lock sync.Mutex来实现多个并发操作的共享数据的互斥锁

package concurrence

import (
	"sync"
	"time"
)

var (
	x    int64
	lock sync.Mutex
)

//带锁的操作
func addWithLock() {
	for i := 0; i < 2000; i++ {
		lock.Lock()
		x += 1
		lock.Unlock()
	}
}

//不带锁的操作
func addWithoutLock() {
	for i := 0; i < 2000; i++ {
		x += 1
	}
}

func Add() {
	x = 0
	for i := 0; i < 5; i++ {
		go addWithoutLock()
	}
	time.Sleep(time.Second)
	println("WithoutLock:", x)
	x = 0
	for i := 0; i < 5; i++ {
		go addWithLock()
	}
	time.Sleep(time.Second)
	println("WithLock:", x)
}

依赖测试（Mock）

因为存在依赖程序生成测试数据到本地文件来进行对比的单元测试，可能存在本地文件被串改，导致测试不正确

那么可以通过Mock来进行打桩（制造测试数据）

MockTrain.go

package demo

import (
	"bufio"
	"os"
	"strings"
)

//从文件读取数据
func ReadData() string {
	open, err := os.Open("log")
	defer open.Close()
	if err != nil {
		return ""
	}
	scanner := bufio.NewScanner(open)
	for scanner.Scan() {
		return scanner.Text()
	}
	return ""
}
//对文件数据进行替换
func ProcessFile() string {
	line := ReadData()
	destLine := strings.ReplaceAll(line, "11", "00")
	return destLine
}

MockTrain_test.go

package demo

import (
	"github.com/stretchr/testify/assert"
	"testing"
)

func TestProcessFile(t *testing.T) {
	line := ProcessFile()
	assert.Equal(t, "line00", line)
}

上述这样生成数据依赖于本地的 log 文件

下面使用Mock来实现生成测试数据

这样测试数据就在程序内部生成，不强依赖外部的文件

MockTrain_test.go

package demo

import (
	"bou.ke/monkey" //依赖包
	"github.com/stretchr/testify/assert"
	"testing"
)

func TestProcessFileWithMock(t *testing.T) {
	//开启，重写生成测试数据的函数
	monkey.Patch(ReadData, func() string {
		return "line120"
	})
	//关闭
	defer monkey.Unpatch(ReadData)
	line := ProcessFile()
	assert.Equal(t, "line000", line)
}

balence.go

package benchmark

import (
	"github.com/bytedance/gopkg/lang/fastrand"
	"math/rand"
)

var ServerIndex [10]int

func InitServerIndex() {
	for i := 0; i < 10; i++ {
		ServerIndex[i] = i+100
	}
}

//使用rand
func Select() int {
	return ServerIndex[rand.Intn(10)]
}

//使用fastrand
func FastSelect() int {
	return ServerIndex[fastrand.Intn(10)]
}

balence_test.go

package benchmark

import (
	"testing"
)

//基准测试开头是 Benchmark
//串行 + rand 
func BenchmarkSelect(b *testing.B) {
	InitServerIndex()
	//初始化计时器 ,从当前开始计时
	b.ResetTimer()
	for i := 0; i < b.N; i++ {
		Select()
	}
}

//并行 + rand
func BenchmarkSelectParallel(b *testing.B) {
	InitServerIndex()
	b.ResetTimer()
	b.RunParallel(func(pb *testing.PB) {
		for pb.Next() {
			Select()
		}
	})
}

//并行 + fastrand
func BenchmarkFastSelectParallel(b *testing.B) {
	InitServerIndex()
	b.ResetTimer()
	b.RunParallel(func(pb *testing.PB) {
		for pb.Next() {
			FastSelect()
		}
	})
}

命名规范

变量

  • 缩略词全部大写，位于变量开头且不需要导出的时候，使用全小写
    • 例如 ServeHTTP 而不是ServeHttp
    • 使用 XMLHTTPRequest 或者 xmlHTTPRequest
  • 变量距离被使用的地方越远，越需要越多的上下文信息
    • 全局变量

函数

  • 函数的参数名尽可能的有特定的含义，减少信息量
  • 函数名尽量不携带包名的上下文
    • http包中有Serve和ServeHTTP函数（命名应该是前者）
    • 使用 http.Serve，而不是http.ServeHTTP

package

  • 全部由小写字母

  • 简短且具有上下文信息

  • 不要和标准库同名

  • 不要使用常用变量名作为包名

  • 使用单数而不是复数

  • 谨慎使用缩写

小结

  • 目标就是降低阅读理解代码成本
  • 重点考虑上下文信息，设计简洁清晰的名称

控制流程

  • 优先处理错误情况，尽早返回减少嵌套
  • 尽量保持最小缩进，避免复杂嵌套分支

  • panic用于真正的异常
